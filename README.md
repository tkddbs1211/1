#include <stdio.h>

int main()
{

	int i = -1;
	/*i값을 -1로 둠 컴퓨터가 2의 보수를 사용한다면, 11111111 ...이 나오게 됨.
	하지만 1의 보수를 사용한다면, 11111110...이 나오게 될것임.*/

	if ((unsigned int)i == ~0)
	{	
		/*
		~0은 00000000...이기에 not 연산자로 11111111....로 바꿈,
		i값도 unsigned int를 사용해 해석 방식을 바꿔줌.
		비교했을 때 둘이 같으면 2의 보수임 
		*/
		printf("과제 1) 출력\n");
		printf("2의 보수입니다.");
	}
	else
	{	
		printf("과제 1) 출력\n");
		printf("1의 보수입니다.");
	}

	return 0;

}
